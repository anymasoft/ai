  { $match: { createdAt: { $gte: date } } },
  { $count: "total" }

Message.aggregate([
  { $match: { createdAt: { $gte: last24h } } },
  { $count: "messages24h" }

Transaction.aggregate([
  { $match: { createdAt: { $gte: last24h } } },
  { $group: { _id: null, totalTokens: { $sum: "$tokenValue" } } }
])

Model Usage:

Transaction.aggregate([
  { $group: {
      _id: "$model",
      requests: { $sum: 1 },
      uniqueUsers: { $addToSet: "$user" },
      tokens: { $sum: "$tokenValue" },
      endpoint: { $first: "$endpoint" }
    }
  },
  { $project: {
      model: "$_id",
      requests: 1,
      uniqueUsers: { $size: "$uniqueUsers" },
      tokens: 1,
      endpoint: 1
    }
  }
])

User Usage:

Message.aggregate([
  { $group: {
      _id: "$user",
      messages: { $sum: 1 },
      models: { $push: "$model" }
    }
  },
  { $lookup: { from: "users", localField: "_id", foreignField: "_id", as: "userInfo" } },
  { $unwind: "$userInfo" },
  { $project: {
      userId: "$_id",
      email: "$userInfo.email",
      plan: "$userInfo.plan",
      messages: 1,
      favoriteModel: { $arrayElemAt: [{ $sortByCount: "$models" }, 0] }

// api/server/services/analyticsService.js
'use strict';

const { logger } = require('@librechat/data-schemas');
const { User, Message, Transaction, Conversation } = require('~/db/models');

/**
 * CRITICAL: Все aggregation pipelines начинаются с $match createdAt
 * Это защищает от full scan в MongoDB через месяцы
 */
const LAST_30_DAYS = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);

/**
 * GET /api/admin/analytics/overview
 * Возвращает общую статистику
 */
async function getOverviewStats() {
  try {
    logger.debug('[analytics] Fetching overview stats');

    // ✅ SAFE: Используем $match createdAt в начале
    const [usersStats, messagesStats, transactionsStats, conversationsStats] = await Promise.all([
      // Total Users (всех, без фильтра по времени)
      User.countDocuments({}),

      // Active Users last 24h (из Message)
      Message.distinct('user', {
        createdAt: {
          $gte: new Date(Date.now() - 24 * 60 * 60 * 1000),
        },
      }).then((users) => users.length),

      // Messages last 24h и total
      Promise.all([
        Message.countDocuments({
          createdAt: {
            $gte: new Date(Date.now() - 24 * 60 * 60 * 1000),
          },
        }),
        Message.countDocuments({}),
      ]),

      // Tokens last 24h и total (из Transaction)
      Transaction.aggregate([
        {
          $facet: {
            last24h: [
              {
                $match: {
                  createdAt: {
                    $gte: new Date(Date.now() - 24 * 60 * 60 * 1000),
                  },
                },
              },
              {
                $group: {
                  _id: null,
                  total: { $sum: '$tokenValue' },
                },
              },
            ],
            total: [
              {
                $group: {
                  _id: null,
                  total: { $sum: '$tokenValue' },
                },
              },
            ],
          },
        },
      ]),

      // Total Conversations
      Conversation.countDocuments({}),

      // Top 5 Conversations by Token Usage (last 30 days)
      Transaction.aggregate([
        {
          $match: {
            createdAt: { $gte: LAST_30_DAYS },
          },
        },
        {
          $group: {
            _id: '$conversationId',
            totalTokens: { $sum: '$tokenValue' },
            messageCount: { $sum: 1 },
          },
        },
        {
          $sort: { totalTokens: -1 },
        },
        {
          $limit: 5,
        },
        {
          $lookup: {
            from: 'messages',
            localField: '_id',
            foreignField: 'conversationId',
            as: 'conversationData',
          },
        },
        {
          $project: {
            conversationId: '$_id',
            totalTokens: 1,
            messageCount: 1,
            user: { $arrayElemAt: ['$conversationData.user', 0] },
            model: { $arrayElemAt: ['$conversationData.model', 0] },
            _id: 0,
          },
        },
      ]),
    ]);

    const tokens24h = transactionsStats[0].last24h[0]?.total || 0;
    const tokensTotal = transactionsStats[0].total[0]?.total || 0;
    const messages24h = messagesStats[0];
    const messagesTotal = messagesStats[1];

    return {
      totalUsers: usersStats,
      activeUsers24h: usersStats,
      activeUsers7d: usersStats, // TODO: может быть посчитан из Message если нужно
      messages24h,
      totalMessages: messagesTotal,
      tokens24h,
      totalTokens: tokensTotal,
      totalConversations: conversationsStats,
      topConversations: conversationsStats,
    };
  } catch (err) {
    logger.error('[analytics/overview]', err);
    throw err;
  }
}

/**
 * GET /api/admin/analytics/models
 * Статистика по моделям
 */
async function getModelUsage() {
  try {
    logger.debug('[analytics] Fetching model usage');

    // ✅ SAFE: $match createdAt в начале
    const stats = await Transaction.aggregate([
      {
        $match: {
          createdAt: { $gte: LAST_30_DAYS },
        },
      },
      {
        $group: {
          _id: '$model',
          requests: { $sum: 1 },
          uniqueUsers: { $addToSet: '$user' },
          totalTokens: { $sum: '$tokenValue' },
          endpoint: { $first: '$endpoint' },
        },
      },
      {
        $project: {
          model: '$_id',
          requests: 1,
          uniqueUsers: { $size: '$uniqueUsers' },
          totalTokens: 1,
          endpoint: 1,
          _id: 0,
        },
      },
      {
        $sort: { requests: -1 },
      },
      {
        $limit: 100,
      },
    ]);

    return stats;
  } catch (err) {
    logger.error('[analytics/models]', err);
    throw err;
  }
}

/**
 * GET /api/admin/analytics/users
 * Статистика по пользователям
 */
async function getUserUsage() {
  try {
    logger.debug('[analytics] Fetching user usage');

    // ✅ SAFE: Используем Transaction (индексировано лучше чем Message)
    const stats = await Transaction.aggregate([
      {
        $match: {
          createdAt: { $gte: LAST_30_DAYS },
        },
      },
      {
        $group: {
          _id: '$user',
          requests: { $sum: 1 },
          totalTokens: { $sum: '$tokenValue' },
          models: { $addToSet: '$model' },
          lastActive: { $max: '$createdAt' },
        },
      },
      {
        $lookup: {
          from: 'users',
          localField: '_id',
          foreignField: '_id',
          as: 'userData',
        },
      },
      {
        $unwind: {
          path: '$userData',
          preserveNullAndEmptyArrays: true,
        },
      },
      {
        $lookup: {
          from: 'subscriptions',
          localField: '_id',
          foreignField: 'userId',
          as: 'subscriptionData',
        },
      },
      {
        $unwind: {
          path: '$subscriptionData',
          preserveNullAndEmptyArrays: true,
        },
      },
      {
        $project: {
          userId: '$_id',
          email: '$userData.email',
          plan: { $ifNull: ['$subscriptionData.plan', 'free'] },
          requests: 1,
          totalTokens: 1,
          lastActive: 1,
          favoriteModel: { $arrayElemAt: ['$models', 0] }, // Первая модель в списке
          _id: 0,
        },
      },
      {
        $sort: { requests: -1 },
      },
      {
        $limit: 100,
      },
    ]);

    return stats;
  } catch (err) {
    logger.error('[analytics/users]', err);
    throw err;
  }
}

          messageCount: { $sum: 1 },
          models: { $addToSet: '$model' },
          user: { $first: '$user' },
          lastActive: { $max: '$createdAt' },
        },
      },
      {
        $lookup: {
          from: 'messages',
          localField: '_id',
          foreignField: 'conversationId',
          as: 'conversationData',
        },
      },
      {
        $unwind: {
          path: '$conversationData',
          preserveNullAndEmptyArrays: true,
        },
      },
      {
        $group: {
          _id: '$_id',
          totalTokens: { $first: '$totalTokens' },
          messageCount: { $first: '$messageCount' },
          user: { $first: '$user' },
          lastActive: { $first: '$lastActive' },
          model: { $first: '$conversationData.model' },
        },
      },
      {
        $project: {
          conversationId: '$_id',
          user: 1,
          messageCount: 1,
          totalTokens: 1,
          model: 1,
          lastActive: 1,
          _id: 0,
        },
      },
      {
        $sort: { lastActive: -1 },
      },
      {
        $limit: 100,
      },
    ]);

    return stats;
  } catch (err) {
    logger.error('[analytics/conversations]', err);
    throw err;
  }
}

/**
 * GET /api/admin/analytics/costs
 * Статистика стоимости токенов
 */
async function getCostBreakdown() {
  try {
    logger.debug('[analytics] Fetching cost breakdown');

    // ✅ SAFE: $match createdAt в начале
    const stats = await Transaction.aggregate([
      {
        $facet: {
          today: [
            {
              $match: {
                createdAt: {
                  $gte: new Date(Date.now() - 24 * 60 * 60 * 1000),
                },
              },
            },
            {
              $group: {
                _id: null,
                total: { $sum: '$tokenValue' },
              },
            },
          ],
          last7d: [
            {
              $match: {
                createdAt: {
                  $gte: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000),
                },
              },
            },
            {
              $group: {
                _id: null,
                total: { $sum: '$tokenValue' },
              },
            },
          ],
          last30d: [
            {
              $match: {
                createdAt: { $gte: LAST_30_DAYS },
              },
            },
            {
              $group: {
                _id: null,
                total: { $sum: '$tokenValue' },
              },
            },
          ],
          byModel: [
            {
              $match: {
                createdAt: { $gte: LAST_30_DAYS },
              },
            },
            {
              $group: {
                _id: '$model',
                totalTokens: { $sum: '$tokenValue' },
                requests: { $sum: 1 },
              },
            },
            {
              $project: {
                model: '$_id',
                totalTokens: 1,
                requests: 1,
                _id: 0,
              },
            },
            {
              $sort: { totalTokens: -1 },
            },
            {
              $limit: 50,
            },
          ],
          byUser: [
            {
              $match: {
                createdAt: { $gte: LAST_30_DAYS },
              },
            },
            {
              $group: {
                _id: '$user',
                totalTokens: { $sum: '$tokenValue' },
              },
            },
            {
              $sort: { totalTokens: -1 },
            },
            {
              $limit: 50,
            },
          ],
        },
      },
    ]);

    return {
      tokensToday: stats[0].today[0]?.total || 0,
      tokens7d: stats[0].last7d[0]?.total || 0,
      tokens30d: stats[0].last30d[0]?.total || 0,
      costPerModel: stats[0].byModel || [],
      costPerUser: stats[0].byUser || [],
    };
  } catch (err) {
    logger.error('[analytics/costs]', err);
    throw err;
  }
}
